/api/game/new POST {"id": <string>} -- start a new game or join existing one
/api/game/stop POST {"id": <string>} -- stop and existing game
/api/game/next POST {"id": <string>} -- finish current turn
/api/game/state GET {"id": <string>} -- get the game state
/api/game/play POST {"id": <string>, "card": <string>} -- play a card (put on table)
/api/game/attack POST {"id": <string>, "card": <string>, "target": <string>} -- attack target using a card
